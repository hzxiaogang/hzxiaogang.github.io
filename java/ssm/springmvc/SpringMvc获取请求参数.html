3. @RequestParam жіЁи§ЈиҺ·еҸ– вҖӢ

@RequestParam жҳҜе°ҶиҜ·жұӮеҸӮж•°е’ҢжҺ§еҲ¶еҷЁж–№жі•зҡ„еҪўеҸӮеҲӣе»әжҳ е°„е…ізі»гҖӮ@RequestParamжіЁи§ЈжңүдёүдёӘеұһжҖ§пјҡ

3.1 value вҖӢ

жҢҮе®ҡдёәеҪўеҸӮиөӢеҖјзҡ„иҜ·жұӮеҸӮж•°зҡ„еҸӮж•°еҗҚ

3.2 required вҖӢ

и®ҫзҪ®жҳҜеҗҰеҝ…йЎ»дј иҫ“жӯӨиҜ·жұӮеҸӮж•°пјҢй»ҳи®ӨеҖјдёә trueгҖӮ

иӢҘи®ҫзҪ®дёә true ж—¶ вҖӢ

еҲҷеҪ“еүҚиҜ·жұӮеҝ…йЎ»дј иҫ“value жүҖжҢҮе®ҡзҡ„иҜ·жұӮеҸӮж•°пјҢиӢҘжІЎжңүдј иҫ“иҜҘиҜ·жұӮеҸӮж•°пјҢдё”жІЎжңүи®ҫзҪ® defaultValue еұһжҖ§пјҢеҲҷйЎөйқўжҠҘй”ҷ 400пјҡRequired String parameter 'xxx' is not presentпјӣ

и®ҫзҪ®дёә false ж—¶ вҖӢ

еҲҷеҪ“еүҚиҜ·жұӮдёҚжҳҜеҝ…йЎ»дј иҫ“ value жүҖжҢҮе®ҡзҡ„иҜ·жұӮеҸӮж•°пјҢиӢҘжІЎжңүдј иҫ“пјҢеҲҷжіЁи§ЈжүҖж ҮиҜҶзҡ„еҪўеҸӮзҡ„еҖјдёәnull

3.3 defaultValue вҖӢ

дёҚз®Ў required еұһжҖ§еҖјдёә true жҲ– falseпјҢеҪ“ value жүҖжҢҮе®ҡзҡ„иҜ·жұӮеҸӮж•°жІЎжңүдј иҫ“жҲ–дј иҫ“зҡ„еҖјдёә""ж—¶пјҢеҲҷдҪҝз”Ёй»ҳи®ӨеҖјдёәеҪўеҸӮиөӢеҖјгҖӮ

7. иҺ·еҸ–иҜ·жұӮеҸӮж•°зҡ„д№ұз Ғй—®йўҳ вҖӢ

1. еҺҹз”ҹ servlet и§ЈеҶід№ұз Ғ вҖӢ

еңЁдҪҝз”Ёservletзҡ„ж—¶еҖҷпјҢжҲ‘д»¬и§ЈеҶіиҜ·жұӮеҸӮж•°д№ұз Ғзҡ„ж–№ејҸжҳҜrequest.setCharacterEncoding("utf-8");пјҢдҪҶжҳҜи®ҫзҪ®иҜ·жұӮзј–з ҒжңүдёҖдёӘиҰҒжұӮпјҢйңҖиҰҒеңЁиҺ·еҸ–иҜ·жұӮеҸӮж•°д№ӢеүҚи®ҫзҪ®зј–з ҒпјҢдёҖж—ҰиҺ·еҸ–еҲ°еҸӮж•°пјҢеҶҚи®ҫзҪ®зј–з ҒеҲҷж— ж•ҲгҖӮ

иҷҪз„¶жҲ‘д»¬еҸҜд»ҘйҖҡиҝҮservletApiж–№ејҸиҺ·еҸ–еҸӮж•°пјҲHttpServletRequest requestпјүпјҢдҪҶжҳҜиҝҷдёӘrequestжҳҜDispatchServletз»ҷжҲ‘д»¬зҡ„пјҢжҳҜиҺ·еҸ–еҲ°иҜ·жұӮеҸӮж•°д№ӢеҗҺзҡ„requestпјҢжүҖд»ҘеҶҚи®ҫзҪ®setCharacterEncodingжҳҜдёҚз”ҹж•Ҳзҡ„пјҲйңҖиҰҒеңЁиҺ·еҸ–иҜ·жұӮеҸӮж•°еүҚи®ҫзҪ® setCharacterEncodingпјүгҖӮ

2.йҒҮеҲ°зҡ„й—®йўҳ вҖӢ

дёӢйқўжҲ‘д»¬д»ҘжөӢиҜ•жЎҲдҫӢ пјҡйҖҡиҝҮ POJO иҺ·еҸ–еҸӮж•°жқҘжөӢиҜ•еңЁдёҚеҗҢtomcatзүҲжң¬дёӢдёӯж–Үд№ұз Ғжғ…еҶө

    1. еңЁtomcat7,getиҜ·жұӮд№ұз ҒпјҢеҸҜд»ҘеңЁserver.xmlи®ҫзҪ®URIEncoding="UTF-8"и§ЈеҶі
xml
    <Connector port="8080" URIEncoding="UTF-8" protocol="HTTP/1.1"
               connectionTimeout="20000"
               redirectPort="8443" />
    1. еңЁtomcat8пјҢgetиҜ·жұӮдёҚд№ұз ҒгҖӮtomcat8и§ЈеҶідәҶgetиҜ·жұӮд№ұз Ғй—®йўҳ
    1. е°ҶиЎЁеҚ•formзҡ„жҸҗдәӨж–№ејҸж”№жҲҗpostпјҢжҲ‘д»¬еҸ‘зҺ°tomcat7е’Ңtomcat8йғҪдјҡд№ұз Ғ

жүҖд»ҘжҲ‘д»¬дё»иҰҒи§ЈеҶізҡ„жҳҜPOSTиҜ·жұӮд№ұз Ғзҡ„й—®йўҳгҖӮ

springmvc и§ЈеҶіиҜ·жұӮеҸӮж•°д№ұз Ғ вҖӢ

дҪҝз”Ё SpringMVC жҸҗдҫӣзҡ„зј–з ҒиҝҮж»ӨеҷЁ CharacterEncodingFilterпјҢдҪҶжҳҜеҝ…йЎ»еңЁ web.xml дёӯиҝӣиЎҢжіЁеҶҢгҖӮ

жіЁ пјҡ SpringMVC дёӯеӨ„зҗҶзј–з Ғзҡ„иҝҮж»ӨеҷЁдёҖе®ҡиҰҒй…ҚзҪ®еҲ°е…¶д»–иҝҮж»ӨеҷЁд№ӢеүҚпјҢеҗҰеҲҷж— ж•Ҳ

xml
<?xml version="1.0" encoding="UTF-8"?>
<web-app xmlns="http://xmlns.jcp.org/xml/ns/javaee"
         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://xmlns.jcp.org/xml/ns/javaee
         http://xmlns.jcp.org/xml/ns/javaee/web-app_4_0.xsd"
         version="4.0">
    <!--й…ҚзҪ®springзҡ„зј–з ҒиҝҮж»ӨеҷЁ-->
    <filter>
        <filter-name>CharacterEncodingFilter</filter-name>
        <filter-class>org.springframework.web.filter.CharacterEncodingFilter</filter-class>
         <!--и®ҫзҪ®иҜ·жұӮзҡ„зј–з Ғ-->
        <init-param>
            <param-name>encoding</param-name>
            <param-value>UTF-8</param-value>
        </init-param>
         <!--и®ҫзҪ®иҜ·жұӮе’Ңзӣёеә”зҡ„зј–з Ғ-->
        <init-param>
            <param-name>forceEncoding</param-name>
            <param-value>true</param-value>
        </init-param>
    </filter>
    <filter-mapping>
        <filter-name>CharacterEncodingFilter</filter-name>
        <url-pattern>/*</url-pattern>
    </filter-mapping>
    <servlet>
        <servlet-name>springmvc</servlet-name>
        <servlet-class>org.springframework.web.servlet.DispatcherServlet</servlet-class>
        <!-- йҖҡиҝҮеҲқе§ӢеҢ–еҸӮж•°жҢҮе®ҡSpringMVCй…ҚзҪ®ж–Үд»¶зҡ„дҪҚзҪ®е’ҢеҗҚз§° -->
        <init-param>
            <param-name>contextConfigLocation</param-name>
            <!-- дҪҝз”Ёclasspath:иЎЁзӨәд»Һзұ»и·Ҝеҫ„жҹҘжүҫй…ҚзҪ®ж–Үд»¶пјҢдҫӢеҰӮmavenе·ҘзЁӢдёӯзҡ„src/main/resources -->
            <param-value>classpath:springmvc.xml</param-value>
        </init-param>
        <!--
            дҪңдёәжЎҶжһ¶зҡ„ж ёеҝғз»„д»¶пјҢеңЁеҗҜеҠЁиҝҮзЁӢдёӯжңүеӨ§йҮҸзҡ„еҲқе§ӢеҢ–ж“ҚдҪңиҰҒеҒҡ
            иҖҢиҝҷдәӣж“ҚдҪңж”ҫеңЁз¬¬дёҖж¬ЎиҜ·жұӮж—¶жүҚжү§иЎҢдјҡдёҘйҮҚеҪұе“Қи®ҝй—®йҖҹеәҰ
            еӣ жӯӨйңҖиҰҒйҖҡиҝҮжӯӨж Үзӯҫе°ҶеҗҜеҠЁжҺ§еҲ¶DispatcherServletзҡ„еҲқе§ӢеҢ–ж—¶й—ҙжҸҗеүҚеҲ°жңҚеҠЎеҷЁеҗҜеҠЁж—¶
        -->
        <load-on-startup>1</load-on-startup>
    </servlet>
    <servlet-mapping>
        <servlet-name>springmvc</servlet-name>
        <url-pattern>/*</url-pattern>
    </servlet-mapping>
</web-app>

зҺ°еңЁжҲ‘д»¬е°ҶformиЎЁеҚ•зҡ„иҜ·жұӮж–№ејҸж”№дёәpostжҸҗдәӨпјҢжҸҗдәӨиЎЁеҚ•пјҢжҲ‘д»¬еҸ‘зҺ°дёӯж–Үжү“еҚ°жӯЈеёёпјҢд№ұз Ғи§ЈеҶігҖӮ
